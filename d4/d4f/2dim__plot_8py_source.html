55 def round_to_divisable(value, divisable):
56  if value == 0:
57  return 0
58  temp = np.abs(value)
59  a = 0
60  while temp < divisable:
61  temp *= 10.0
62  a += 1
63  if temp % divisable == 0:
64  return value
65  res = (temp - (temp % divisable) + divisable) / (10.0**a)
66  if value < 0:
67  return -res
