121  template <typename T>
122  static std::vector<int> calc_num_spikes_vec(
123  const cypress::Matrix<T> &spiketrains, const cypress::Real start = 0.0,
124  const cypress::Real end = std::numeric_limits<cypress::Real>::max())
125  {
126  std::vector<int> res;
127  size_t rows = spiketrains.rows();
128  size_t colls = spiketrains.cols();
129  for (size_t i = 0; i < rows; i++) {
130  int counter = 0;
131  for (size_t j = 0; j < colls; j++) {
132  if ((spiketrains[i * colls + j] >= start - 0.001) &&
133  (spiketrains[i * colls + j] <= end + 0.001)) {
134  counter++;
135  }
136  }
137  res.push_back(counter);
138  }
139  return res;
140  }
